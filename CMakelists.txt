cmake_minimum_required(VERSION 3.16.0)

project(LearnUSB VERSION 1.0.0 LANGUAGES CXX C)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTORCC ON)
set(CMAKE_AUTOUIC ON)

find_package(Qt6 COMPONENTS Widgets REQUIRED)

set(THIRD_PARTY_DIR "E:\\usb\\binary")

include_directories(${THIRD_PARTY_DIR}/include/libusb-1.0)
link_directories(${THIRD_PARTY_DIR}/VS2019/MS64/dll)

add_executable(testlibusb
    testlibusb.c
)
add_executable(listdevs
        listdevs.c
        )
add_executable(xusb
        xusb.c
        )

target_link_libraries(testlibusb libusb-1.0)
target_link_libraries(listdevs libusb-1.0)
target_link_libraries(xusb libusb-1.0)
